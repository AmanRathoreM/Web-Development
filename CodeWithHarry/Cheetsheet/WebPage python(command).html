    # Let's create a dictionary with a name dict1

dict2 = dict1.copy()
# Now if you change the value of dict1 the value of dict2 will absolutely not change 
# But if you had not used .copy() then value will absolutely change  

dict1.update({"Sorab":"Ice Creame"})

print("The value of dict1 is:",dict1)
print("\nAnd the value of dict2 is:",dict2)
